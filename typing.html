/* styles.css */ body { font-family: 'Arial', sans-serif; display: flex; justify-content: center; align-items: center; height: 100vh; background-color: #f0f0f0; margin: 0; } #game-container { text-align: center; background-color: #ffffff; padding: 20px; box-shadow: 0 0 10px rgba(0,0,0,0.1); } #sushi-container { margin: 20px 0; font-size: 24px; } #typing-input { padding: 10px; font-size: 16px; width: 100%; box-sizing: border-box; } #message { margin-top: 10px; color: #ff6347; } // script.js document.addEventListener('DOMContentLoaded', () => { const sushiList = ['まぐろ', 'サーモン', 'えび', 'いくら', 'たこ', 'ほたて']; const sushiContainer = document.getElementById('sushi-container'); const typingInput = document.getElementById('typing-input'); const message = document.getElementById('message'); let currentSushi = ''; function displayRandomSushi() { const randomIndex = Math.floor(Math.random() * sushiList.length); currentSushi = sushiList[randomIndex]; sushiContainer.textContent = currentSushi; } typingInput.addEventListener('input', () => { if (typingInput.value === currentSushi) { message.textContent = '正解！次の寿司を入力してね！'; typingInput.value = ''; displayRandomSushi(); } else { message.textContent = '間違い... もう一度試してね。'; } }); displayRandomSushi(); });
